% For windows PC and Mac without GPU
1) install Anaconda
2) run the following command at terminal
	git clone https://git.embl.de/ries/psfmodelling.git
	git check out dev_sheng
	conda create --name psfinv python=3.9.6
	conda activate psfinv
	pip install -e psfmodelling


%For windows PC with GPU
1) install Anaconda
2) run the following command at terminal
	git clone https://git.embl.de/ries/psfmodelling.git
	git check out dev_sheng
	conda env create --name psfinv --file=psfmodelling/environment.yml
	conda activate psfinv
	pip install -e psfmodelling


%For linux with GPU
1) install miniconda
2) run the following command at terminal
	git clone https://git.embl.de/ries/psfmodelling.git
	git check out dev_sheng
	conda env create --name psfinv --file=psfmodelling/environment.yml
	conda activate psfinv
	pip install -e psfmodelling	
3) after installation run the following command to add cudnn path
	mkdir -p $CONDA_PREFIX/etc/conda/activate.d
	echo 'export LD_LIBRARY_PATH=$LD_LIBRARY_PATH:$CONDA_PREFIX/lib/' >> $CONDA_PREFIX/etc/conda/activate.d/env_vars.sh



% test on Matlab, 
	- open Matlab script at psfmodelling/examples/run_python_xxx.m
	- set envpath to your conda environment path
	- set runpath to the path of run_python_xxx.m
	- run the script, you should see learning info in the command window, at the end you will see two figures pop up, close those figures, the result .mat file will be saved in the datapath.

% test on terminal,
	- open psfmodelling/examples/params.json, change 'Z' in "datapath" and "savename" to the drive to t2ries on your computer. 
	- open Anaconda Powershell (PC) or terminal (mac)
	- run the following command
		conda activate psfmodelling
		cd yourpath/psfmodelling/examples/
		python learn_psf.py params.json
